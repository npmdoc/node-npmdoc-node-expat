function node-expat.Parser.prototype.end (data)

  • description and source-code
    end = function (data) {
  var error, result
  try {
    result = this.parse(data || '', true)
    if (!result) {
      error = this.getError()
    }
  } catch (e) {
    error = e
  }

  if (!error) {
    this.emit('end')
  } else {
    this.emit('error', error)
  }
  this.emit('close')
}
